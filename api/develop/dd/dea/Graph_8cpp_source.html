151 void Graph::removeNode(AbstractNode& node) {
152  mSyncStorage.erase(node.getNodeIndex());
153  size_t nodeIndex = node.getNodeIndex();
154  auto removePredicate = [nodeIndex](const inner::Edge& edge) -> bool {
155  return nodeIndex == edge.startNodeIndex ||
156  nodeIndex == edge.endNodeIndex;
157  };
158  mTopology.erase(
159  std::remove_if(mTopology.begin(), mTopology.end(), removePredicate),
160  mTopology.end());
161  inner::setTraversalValidity(mTraversal, false);
162 }

205 const Traversal& Graph::traverse() {
206  if (mTraversal.isValidTraversal()) {
207  return mTraversal;
208  }
209  inner::getClusters(mTraversal).clear();
210  std::sort(mTopology.begin(), mTopology.end());
211  std::queue<size_t> currentQueue, newQueue;
212  std::unordered_map<size_t, inner::NodeState> visits;
213  initVisits(visits, mOwningStorage, mSyncStorage);
214  initQueue(currentQueue, mOwningStorage, mSyncStorage);
215  while (true) {
216  inner::Cluster cluster{0};
217  while (!currentQueue.empty()) {
218  size_t nodeIndex = currentQueue.front();
219  currentQueue.pop();
220  inner::Edge target(nodeIndex, 0, 0);
221  auto edgeIterator =
222  std::lower_bound(mTopology.begin(), mTopology.end(), target);
223  while (edgeIterator != mTopology.end() &&
224  edgeIterator->startNodeIndex == nodeIndex) {
225  auto& inputCount =
226  visits[edgeIterator->endNodeIndex].inputCount;
227  ++inputCount;
228  auto targetInputCount =
229  inner::getNodeTable()[edgeIterator->endNodeIndex]
230  ->getInputsCount();
231  if (inputCount == targetInputCount) {
232  newQueue.push(edgeIterator->endNodeIndex);
233  } else if (inputCount > targetInputCount) {
234  FatalError(1, "traverse() in Graph: ", mGraphIndex,
235  ". Graph is have an cycle(s)");
236  }
237  visits[edgeIterator->endNodeIndex].input.emplace_back(
238  nodeIndex, edgeIterator->mark);
239  visits[nodeIndex].output.emplace_back(
240  edgeIterator->endNodeIndex, edgeIterator->mark);
241  ++edgeIterator;
242  }
243  AbstractNode* node = inner::getNodeTable()[nodeIndex];
244  switch (node->getType()) {
245  case NodeType::DEFAULT:
246  cluster.get<Node>().emplace_back(
247  node->getNodeIndex(),
248  std::move(visits[nodeIndex].input),
249  std::move(visits[nodeIndex].output));
250  break;
251  case NodeType::INPUT:
252  cluster.get<InputNode>().emplace_back(
253  node->getNodeIndex(),
254  std::move(visits[nodeIndex].input),
255  std::move(visits[nodeIndex].output));
256  break;
257  default:
258  FatalError(1, "Undefined NodeType in traverse()");
259  }
260  ++cluster.nodeCount;
261  }
262  if (cluster.nodeCount > 0) {
263  inner::getClusters(mTraversal).emplace_back(std::move(cluster));
264  }
265  std::swap(currentQueue, newQueue);
266  if (currentQueue.empty()) {
267  break;
268  }
269  }
270  inner::setTraversalValidity(mTraversal, true);
271  return mTraversal;
272 }
