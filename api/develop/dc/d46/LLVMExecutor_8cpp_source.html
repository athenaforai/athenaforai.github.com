30 void LLVMExecutor::prepare(athena::core::Graph &graph) {
31  mMainModule = std::make_unique<::llvm::Module>("AthenaMain",
32  mJITCompiler->getContext());
33  mMainModule->setDataLayout(mJITCompiler->getDataLayout());
34 
35  // todo consider initializing some optimizers
36 
37  ::llvm::FunctionType *FT = ::llvm::FunctionType::get(
38  ::llvm::Type::getVoidTy(mJITCompiler->getContext()), false);
39  ::llvm::Function::Create(FT, ::llvm::Function::ExternalLinkage, "jitmain",
40  *mMainModule);
41 
42  LLVMGenerator generator(mJITCompiler->getContext(), mMainModule,
43  *mAllocator);
44 
45  mGraphTraversal = graph.traverse();
46 
47  for (auto &cluster : mGraphTraversal.getClusters()) {
48  auto &inputNodes = cluster.get<core::InputNode>();
49  for (auto &nodeDeps : inputNodes) {
50  auto &inputNode = static_cast<core::InputNode &>(
51  *core::inner::getNodeTable()[nodeDeps.nodeIndex]);
52  generator.openNode(inputNode.getName());
53  generator.generate("allocate",
54  core::inner::getTensorFromNode(inputNode));
55  // todo generate code for loader
56  generator.generateLoad(inputNode.getLoader(),
57  core::inner::getTensorFromNode(inputNode));
58  generator.closeNode();
59  }
60 
61  auto &actionNodes = cluster.get<core::Node>();
62  for (auto &nodeDeps : actionNodes) {
63  std::vector<core::inner::Tensor *> preparedTensors;
64  for (auto &input : nodeDeps.input) {
65  auto *node = core::inner::getNodeTable()[input.nodeIndex];
66  preparedTensors.push_back(
67  &core::inner::getTensorFromNode(*node));
68  }
69  auto &node = static_cast<core::Node &>(
70  *core::inner::getNodeTable()[nodeDeps.nodeIndex]);
71  generator.openNode(node.getName());
72  generator.generate("allocate",
73  core::inner::getTensorFromNode(node));
74  preparedTensors.push_back(&core::inner::getTensorFromNode(node));
75  // todo lock tensors in memory
76  node.getOperation().gen(generator, preparedTensors);
77  // todo unlock tensors in memory
78 
79  for (size_t argNo = 0; argNo < nodeDeps.input.size(); argNo++) {
80  // todo check for frozen nodes
81  auto derivativeTensor = node.getOperation().getDerivativeTensor(
82  preparedTensors, argNo);
83  core::inner::addDerivativeTensor(node, *derivativeTensor);
84  preparedTensors.pop_back();
85  preparedTensors.push_back(derivativeTensor);
86  generator.generate("allocate", *derivativeTensor);
87  node.getOperation().genDerivative(generator, preparedTensors,
88  argNo);
89  }
90 
91  generator.closeNode();
92  }
93  }
94 
95  auto builder = generator.getBuilder();
96 
97  builder.CreateRetVoid();
98 }

100 void LLVMExecutor::execute() {
101  auto err = mJITCompiler->addModule(mMainModule);
102  if (err) {
103  core::FatalError(1, "Error adding module to JIT");
104  }
105 
106  auto sym = mJITCompiler->lookup("jitmain");
107 #ifdef DEBUG
108  assert(sym && "Failed to find jitmain function");
109 #endif
110  auto mainFunction = (void (*)())(intptr_t)sym.get().getAddress();
111  mainFunction();
112 }
