42 ::llvm::Value *LLVMGenerator::generateGetFastPointer(core::inner::Tensor &t) {
43  ::llvm::Function *calledFunction =
44  mModule->getFunction("athena_get_fast_pointer");
45 
46  if (!calledFunction)
47  calledFunction = impl::create_get_fast_pointer_decl(mContext, *mModule);
48 
49  if (!calledFunction) {
50  core::FatalError(1, "Unknown function referenced");
51  }
52 
53  std::vector<::llvm::Value *> ArgsV;
54  ::llvm::Constant *allocatorConst = ::llvm::ConstantInt::get(
55  ::llvm::Type::getInt64Ty(mContext), (size_t)(&mAllocator));
56  ArgsV.push_back(allocatorConst);
57  ::llvm::Constant *tensorConst = ::llvm::ConstantInt::get(
58  ::llvm::Type::getInt64Ty(mContext), reinterpret_cast<size_t>(&t));
59  ArgsV.push_back(tensorConst);
60  auto callInst = mBuilder.CreateCall(calledFunction, ArgsV);
61  return callInst;
62 }

64 void LLVMGenerator::generateLoad(const core::AbstractLoader &loader,
65  core::inner::Tensor &tensor) {
66  ::llvm::Function *loadFunction =
67  mModule->getFunction(loader.getLoadCName());
68 
69  if (!loadFunction) {
70  std::vector<::llvm::Type *> args(3, ::llvm::Type::getInt64Ty(mContext));
71  ::llvm::FunctionType *FT = ::llvm::FunctionType::get(
72  ::llvm::Type::getVoidTy(mContext), args, false);
73 
74  loadFunction =
75  ::llvm::Function::Create(FT, ::llvm::Function::ExternalLinkage,
76  loader.getLoadCName(), mModule.get());
77  }
78 
79  if (!loadFunction) {
80  core::FatalError(1, "Unknown function referenced");
81  }
82 
83  std::vector<::llvm::Value *> ArgsV;
84  ::llvm::Constant *loaderConst = ::llvm::ConstantInt::get(
85  ::llvm::Type::getInt64Ty(mContext), reinterpret_cast<size_t>(&loader));
86  ArgsV.push_back(loaderConst);
87  ::llvm::Constant *allocatorConst =
88  ::llvm::ConstantInt::get(::llvm::Type::getInt64Ty(mContext),
89  reinterpret_cast<size_t>(&mAllocator));
90  ArgsV.push_back(allocatorConst);
91  ::llvm::Constant *tensorConst = ::llvm::ConstantInt::get(
92  ::llvm::Type::getInt64Ty(mContext), reinterpret_cast<size_t>(&tensor));
93  ArgsV.push_back(tensorConst);
94  mBuilder.CreateCall(loadFunction, ArgsV);
95 }

115 void LLVMGenerator::openNode(std::string_view name) {
116 #ifdef DEBUG
117  assert(mCurrentBlock == mMainBlock && "There is an opened node");
118 #endif
119  ::llvm::FunctionType *FT =
120  ::llvm::FunctionType::get(::llvm::Type::getVoidTy(mContext), false);
121  auto nodeFunction =
122  ::llvm::Function::Create(FT, ::llvm::Function::ExternalLinkage,
123  "node_" + std::string(name), *mModule);
124 
125  mBuilder.CreateCall(nodeFunction);
126 
127  mCurrentBlock = ::llvm::BasicBlock::Create(mContext, "entry", nodeFunction);
128 
129  mBuilder.SetInsertPoint(mCurrentBlock);
